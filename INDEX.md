## How to use
1. **Filter** above or browse by region to open the relevant guide.
2. Review positioning, approach, critical anatomy, and the step-by-step technique (and the linked **approach chapter** for granular corridor detail).
3. Copy the guide and fill the bracketed `[ ]` fields with the patient's specifics — keep patient-identifiable details on your own secure system, **not** in this public repository.

<p class="muted" id="no-hits" style="display:none; text-align:center; padding:18px;">No guides match — try a different term.</p>

<script>
(function () {
  var box = document.getElementById('guide-filter');
  if (!box) return;
  var lists = Array.prototype.slice.call(document.querySelectorAll('.prose ul'));
  var heads = Array.prototype.slice.call(document.querySelectorAll('.prose h2, .prose h3'));
  var noHits = document.getElementById('no-hits');
  box.addEventListener('input', function () {
    var q = this.value.trim().toLowerCase();
    var totalVisible = 0;
    lists.forEach(function (ul) {
      var anyShown = false;
      Array.prototype.slice.call(ul.children).forEach(function (li) {
        if (li.tagName !== 'LI') return;
        var hit = q === '' || li.textContent.toLowerCase().indexOf(q) > -1;
        li.style.display = hit ? '' : 'none';
        if (hit) { anyShown = true; totalVisible++; }
      });
      ul.style.display = anyShown ? '' : 'none';
    });
    heads.forEach(function (h) {
      if (q === '') { h.style.display = ''; return; }
      var vis = false, n = h.nextElementSibling;
      while (n && !/^H[23]$/.test(n.tagName)) {
        if (n.tagName === 'UL' && n.style.display !== 'none') { vis = true; break; }
        n = n.nextElementSibling;
      }
      h.style.display = vis ? '' : 'none';
    });
    if (noHits) noHits.style.display = (q !== '' && totalVisible === 0) ? 'block' : 'none';
  });
})();
</script>
